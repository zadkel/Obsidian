def extract_next_episode_number(html_content):
    """
    Returns:
        int: 추출된 다음 에피소드 번호. 찾지 못하면 -1을 반환합니다.
    """
    next_episode_number = -1 # 기본값 -1로 초기화
    if not html_content:
        print("오류: HTML 내용이 비어 있습니다.") # 필요 시 주석 해제
        return next_episode_number
        
    # 정규 표현식 패턴: check_next_episode_link( 와 ) 사이에 있는 숫자(\d+)를 찾습니다.
    # 숫자 양 옆에 작은 따옴표(')가 있을 수도 있고 없을 수도 있는 경우를 모두 처리합니다.
    pattern = r"check_next_episode_link\('?(\d+)'?\)"
    match = re.search(pattern, html_content)
    
    # 매칭된 경우
    if match:
        try:
            # 첫 번째 캡처 그룹 (숫자 부분)을 정수로 변환하여 저장
            next_episode_number = match.group(1)
            # print(f"함수 내: 다음화 링크 숫자 {next_episode_number} 추출.") # 필요 시 주석 해제
        except (ValueError, IndexError):
            # 숫자로 변환할 수 없거나 그룹이 없는 경우 오류 처리
            print("오류: 추출된 값을 숫자로 변환하는 데 실패했습니다.")
            next_episode_number = -1 # 실패 시 -1 반환
            
    # 매칭되지 않은 경우
    # else:
        # print("정보: HTML 내용에서 'check_next_episode_link(숫자)' 패턴을 찾지 못했습니다.") # 필요 시 주석 해제
    return next_episode_number

def Crawling(url,i):
  #사이트 열기
  driver.get(url)
  load_delay = random.uniform(10.1,18.7) # 랜덤 대기
  print(f"페이지 로드 후 대기: {load_delay:.2f}초")
  time.sleep(load_delay)
  
  #JavaScript가 실행된 후의 HTML 가져오기
  html_source = driver.page_source
  
  #with open(f"크롤링{i}화.html", "w", encoding="utf-8") as file: # html이 필요하면 해제
    #file.write(html_source)
    
  # BeautifulSoup을 사용한 HTML 파싱
  soup = BeautifulSoup(html_source, "html.parser")
  tag_del(soup)

  # 본문 내용이 있는 태그 찾기
  content = soup.find_all("font", class_="line")  # 'font' 태그 중 class가 'line'인 것만 추출

  # 태그를 제거하고 텍스트만 추출
  text_content = "\n".join(tag.get_text(strip=True) for tag in content)

  # 텍스트를 파일로 저장
  with open(f"이세계 컴퓨터 개발자 {i:0>3}화.txt", "w", encoding="utf-8") as f:
      f.write(text_content)
  print(f"📄 {i}화 저장 완료!")

  #다음화 정보 추출
  return extract_next_episode_number(html_source)

def tag_del(soup):
  invisible_tags = soup.find_all("p", style=lambda s: s and "height: 0px" in s and "width: 0px" in s and "overflow: hidden" in s and "opacity: 0" in s)
  if invisible_tags:
      for tag in invisible_tags:
          tag.decompose()
      #print(f"🕵️‍♀️ 방해용 <p> 태그 {len(invisible_tags)}개를 제거했습니다.")
